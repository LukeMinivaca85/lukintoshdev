lukintosh.dev/quickstart
// Install Lukintosh developer tools
npm install @lukintosh/sdk

// Initialize a secure client
import { Lukintosh } from "@lukintosh/sdk";

const client = new Lukintosh({
  app: "my-product",
  environment: "production",
  auth: {
    mfa: true,
    sessions: true,
    loginAlerts: true
  }
});

await client.accounts.verifySession();
await client.security.reportHealth();

// Build with identity, trust, and platform APIs.
